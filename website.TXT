<script>
  function handleSubmit(e) {
    e.preventDefault();
    const form = e.target;
    let data = {
      Name: form.name.value,
      Size: form.size.value,
      JerseyNumber: form.number.value,
      SleeveType: form.sleeve.value,
      PaymentMethod: form.payment.value,
      PaymentNumber: form.paymentNumber.value,
      Signature: form.signature.value
    };
    alert('Thank you for your order!\n\n' + JSON.stringify(data, null, 2));
    form.reset();
    return false;
  }
